# we can see from the variables above that we can categorize those "No xxx(such as internet) service" to "No". For example, if 
# an observation did not subscribe phone service, then he can't have multiplelines service

#Then we will have 14 binary categorical variables, let's first transform them, we assign positive answer as 1, negative as 0.
# For gender, we assign Male as 1 and Female as 0
def cateTransformer(colname):
    for i in df[colname].unique():
        if i == 'Yes':
            df[colname].replace(i,1,inplace=True)
        if i != 'Yes':
            df[colname].replace(i,0,inplace=True)
